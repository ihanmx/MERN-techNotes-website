server {
    listen 80;
    server_name 178.105.138.179;

    location / {
        proxy_pass http://localhost:5000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_cache_bypass $http_upgrade;
    }
}

Directive	Meaning
server { ... }	Defines one "virtual host" — one site nginx serves. You can have many server blocks for different domains.
listen 80;	Listen on port 80 (HTTP). Will become 443 ssl; after Let's Encrypt.
server_name 178.105.138.179;	Which hostnames this block answers to. Right now your IP; later your domain.
location / { ... }	Rules for URL path matching / (which means everything).
proxy_pass http://localhost:5000;	The core of the reverse proxy. Forward this request to Node on port 5000.
proxy_http_version 1.1;	Use HTTP/1.1 to the backend (default is 1.0, which breaks WebSockets and keep-alive).
proxy_set_header Host $host;	Pass the original Host header to Node (so Node sees the real hostname, not localhost).
proxy_set_header X-Real-IP $remote_addr;	Tell Node the client's actual IP (without this, Node sees nginx's IP = 127.0.0.1).
proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;	Append the client IP to the X-Forwarded-For chain (standard for multi-hop proxies).
proxy_set_header X-Forwarded-Proto $scheme;	Tell Node whether the original request was http or https. Important after HTTPS is on.
proxy_set_header Upgrade $http_upgrade; + Connection 'upgrade';	Enable WebSocket support. Not needed for TechNote, but free to leave in.
proxy_cache_bypass $http_upgrade;	Don't serve WebSocket requests from cache. Same — defensive.
